<script>
    const toggleSwitch = document.querySelector('.theme-switch input[type="checkbox"]');
    function switchTheme(e) {
        if (e.target.checked) {
            document.documentElement.setAttribute('data-theme', 'light');
        } else {
            document.documentElement.setAttribute('data-theme', 'dark');
        }    
    }
    toggleSwitch.addEventListener('change', switchTheme, false);

    let rowIdCounter = 0;
    const defaultColors = ["#f28b82", "#fcb92c", "#8ab4f8", "#81c995", "#c58af9", "#9aa0a6"];

    function addRow() {
        rowIdCounter++;
        const list = document.getElementById('subject-list');
        const row = document.createElement('div');
        row.className = 'input-row';
        row.id = `input-row-${rowIdCounter}`;
        const colorValue = defaultColors[(rowIdCounter - 1) % defaultColors.length];
        row.innerHTML = `
            <span class="row-number">${rowIdCounter}</span>
            <input type="text" class="input-field name" placeholder="教科名">
            <input type="number" class="input-field score" placeholder="点数">
            <input type="number" class="input-field max" placeholder="満点" value="100">
            <input type="color" class="input-field color" value="${colorValue}">
            <button class="btn btn-delete" onclick="removeRow('${row.id}')">削除</button>
        `;
        list.appendChild(row);
        updateRowNumbers();
    }

    function removeRow(id) {
        const row = document.getElementById(id);
        if (row) { row.remove(); updateRowNumbers(); }
    }

    function updateRowNumbers() {
        const rows = document.querySelectorAll('.input-row');
        rows.forEach((row, index) => {
            row.querySelector('.row-number').textContent = index + 1;
        });
    }

       function createGraph() {
        const titleInput = document.getElementById('exam-title').value || "模試結果";
        const displaySection = document.getElementById('display');
        const displayTitle = document.getElementById('display-title');
        const graphRowsContainer = document.getElementById('graph-rows');

        graphRowsContainer.innerHTML = '';
        displayTitle.textContent = titleInput;

        const rows = document.querySelectorAll('.input-row');
        let totalScore = 0;
        let totalMax = 0;
        let hasData = false;
        
        rows.forEach(row => {
            const name = row.querySelector('.name').value.trim();
            const scoreStr = row.querySelector('.score').value.trim();
            const maxStr = row.querySelector('.max').value.trim();
            const color = row.querySelector('.color').value;

            // 点数が入力されている場合のみ集計・表示対象にする
            if (scoreStr !== "") {
                hasData = true;
                const score = parseFloat(scoreStr) || 0;
                const max = parseFloat(maxStr) || 100;
                totalScore += score;
                totalMax += max;
                
                let percentage = Math.min((score / max) * 100, 100);

                const graphRow = document.createElement('div');
                graphRow.className = 'graph-row';
                graphRow.innerHTML = `
                    <div class="row-labels">
                        <div class="label-color-bar" style="background-color: ${color};"></div>
                        <div class="label-name">${name || "無題"}</div>
                        <div class="label-score-box">
                            <span class="label-score">${score}</span>
                            <span class="label-max">/${max}</span>
                        </div>
                    </div>
                    <div class="bar-area">
                        <div class="bar" style="width: 0%; background-color: ${color}; color: #fff;">
                            ${Math.round(score)}
                        </div>
                    </div>
                `;
                graphRowsContainer.appendChild(graphRow);
                setTimeout(() => {
                    graphRow.querySelector('.bar').style.width = `${percentage}%`;
                }, 100);
            }
        });

        // 合計行の表示（データがある場合のみ）
        if (hasData && totalMax > 0) {
            let totalPercentage = Math.min((totalScore / totalMax) * 100, 100);
            const totalRow = document.createElement('div');
            totalRow.className = 'graph-row total-row';
            totalRow.innerHTML = `
                <div class="row-labels">
                    <div class="label-color-bar" style="background-color: var(--total-bar-color);"></div>
                    <div class="label-name" style="font-weight: bold;">合計</div>
                    <div class="label-score-box">
                        <span class="label-score">${totalScore}</span>
                        <span class="label-max">/${totalMax}</span>
                    </div>
                </div>
                <div class="bar-area">
                    <div class="bar" style="width: 0%; background-color: var(--total-bar-color);">
                        ${Math.round(totalScore)}
                    </div>
                </div>
            `;
            graphRowsContainer.appendChild(totalRow);
            setTimeout(() => {
                totalRow.querySelector('.bar').style.width = `${totalPercentage}%`;
            }, 150);

            displaySection.style.display = 'block';
            window.scrollTo({ top: displaySection.offsetTop - 20, behavior: 'smooth' });
        }
    }

    addRow(); addRow(); addRow();
</script>
